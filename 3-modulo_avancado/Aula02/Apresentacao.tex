\section{Spatial Separable Convolution}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Spatial Separable Convolution}
\begin{itemize}
\item A {\it spatial separable convolution} é assim chamada porque manipula as dimensões espaciais de uma imagem e do kernel: a largura e a altura.
\item Divide um kernel em dois kernels menores.
\item Como exemplo, considere o filtro de Sobel:
\end{itemize}
\begin{figure}[h]
\includegraphics[width=6cm]{imgs/spatial_separable_convolution_example.png}
\end{figure}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Spatial Separable Convolution}
\begin{figure}[h]
\includegraphics[width=8cm]{imgs/spatial_separable_convolution.png}
\end{figure}

\vspace{1cm}
\tiny{Fonte: \href{https://towardsdatascience.com/a-basic-introduction-to-separable-convolutions-b99ec3102728}{A Basic Introduction to Separable Convolutions}}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Spatial Separable Convolution}
\begin{itemize}
\item Ao invés de fazer 1 convolução com 9 multiplicações, são feitas 2 convoluções com 3 multiplicações (6 no total) cada, que alcançam o mesmo efeito. 
\item Com menos multiplicações, a complexidade computacional diminui e a rede executa mais rapidamente.
\item O principal problema é que com {\it spatial separable convolutions} nem todos os kernels podem ser ``separados'' em dois kernels menores.
\end{itemize}
\end{frame}

\section{Depthwise Separable Convolutions}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Depthwise Separable Convolutions}
\begin{itemize}
\item {\it Depthwise separable convolutions} funcionam com kernels que não podem ser ``fatorados'' em dois kernels menores.
\item Divide um kernel em 2 kernels separados que executam duas convoluções:
\begin{itemize}
\item uma {\it depthwise convolution} e 
\item uma {\it pointwise convolution}.
\end{itemize} 
\end{itemize}
\end{frame}

\begin{frame}{MobileNet}{Versões}
\begin{itemize}
\item {\bf MobileNet V2}: mudanças significativas foram feitas na arquitetura MobileNetV1 que resultaram em um aumento considerável na precisão do modelo. 
\item {\bf MobileNet V3 Small}: é 6,6\% mais precisa na classificação do que MobileNetV2 e tem latência semelhante. 
\item {\bf MobileNet V3 Large}: é 3,2\% mais precisa na classificação e reduz a latência em 20\% quando comparado ao MobileNetV2.
\end{itemize}
