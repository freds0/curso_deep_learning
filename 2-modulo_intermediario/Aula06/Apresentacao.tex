\section{Introdução}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Introdução}{CNN Architectures}
\begin{itemize}
\item A arquitetura básica de um modelo convolucional é composta por uma etapa de {\it extração de features} e outra de {\it classificação}.
\item A fase de {\bf extração de features} é formada por filtros convolucionais com função de ativação e max-pooling.
\item A fase de {\bf classificação} é formada por camadas fully-connected e função de ativação {\it softmax} para classificação.
\end{itemize}
\end{frame}

\subsection{LeNet}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Architectures}{LeNet}
\begin{itemize}
\item A primeira aplicação bem-sucedida de redes convolucionais foi apresentada por Yann LeCun em 1998.
\item Conhecida como {\bf LeNet}, essa rede apresenta 7 camadas, convolucionais e densas.
\item Foi utilizada para reconhecer os números do CEP em correspondências (MNIST).

\vspace{1cm}
\tiny{Referência: \href{http://yann.lecun.com/exdb/publis/pdf/lecun-98.pdf }{Gradient-Based Learning Applied to Document Recognition} }
\end{itemize}
\end{frame}

\subsection{AlexNet}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Architectures}{AlexNet}
\begin{itemize}
\item {\bf AlexNet} é um modelo convolucional proposto em 2012 por Alex Krizhevsky, Ilya Sutskever e Geoffrey Hinton.
\item Sua arquitetura é formada por 8 camadas: 5 camadas convolucionais e 3 camadas fully-connected.
\item ImageNet possui $\approx$ 14 milhões de imagens pertencentes a 1000 classes.
\item O modelo alcançou o melhor 85,7\% de acurácia no Top-5.

\vspace{1cm}
\tiny{Referência: \href{https://papers.nips.cc/paper/2012/file/c399862d3b9d6b76c8436e924a68c45b-Paper.pdf}{ImageNet Classification with Deep Convolutional Neural Networks} }
\end{itemize}
\end{frame}

\subsection{VGG16}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Architectures}{VGG16}
\begin{itemize}
\item {\bf VGG16} é um modelo convolucional proposto em 2014 por K. Simonyan and A. Zisserman.
\item Esse modelo alcançou 92,7\% de acurácia no top-5 no dataset ImageNet.
\item Possui 16 camadas com filtros 3x3 com {\it stride}=1 e {\it padding} seguido de {\it max pooling} 2x2 e 2 camadas {\it fully-connected}.

\vspace{2cm}
\tiny{Referência: \href{https://arxiv.org/abs/1409.1556}{Very Deep Convolutional Networks for Large-Scale Image Recognition} }
\end{itemize}
\end{frame}

\section{Transfer Learning}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Transfer Learning}
\begin{itemize}
\item {\bf Transfer Learning} é uma técnica em que um modelo desenvolvido para uma tarefa é reutilizado como estágio inicial para uma outra tarefa..
\item É amplamente utilizado nas áreas como visão computacional e NLP, devido ao custo alto para treinamento de modelos, 
\item Funciona apenas se as {\it features} aprendidas na primeira tarefa forem também adequadas para a segunda tarefa.
\end{itemize}

\vspace{2cm}
\tiny{Fonte: \href{https://machinelearningmastery.com/transfer-learning-for-deep-learning/}{A Gentle Introduction to Transfer Learning for Deep Learning} }
\end{frame}
