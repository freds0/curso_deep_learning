\section{Introdução}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Convoluçao}
\begin{itemize}
\item {\bf Convoluçao} é a simples aplicação de um filtro sobre uma entrada resultando em uma ativação.
\item A aplicação de forma repetida de um filtro (ou kernel) sobre uma imagem resulta em um {\it mapa de ativação}, também chamado {\bf mapa de features}. 
\item A principal vantagem é que os filtros preservam as relações espaciais entre os pixels.
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Convoluçao}
\begin{itemize}
\item A inovaçao das {\bf Redes Neurais Convolucionais} (CNNs) é a habilidade de automaticamente aprender um grande número de filtros.
\item Camadas convolucionais podem conter diversos filtros, que são aplicados em paralelo.
\item Como resultado, obtém-se diversos filtros especializados em extrair {\it features} específicas para determinadas tarefas.
\end{itemize}

\vspace{1cm}
\tiny{Fonte: \href{https://machinelearningmastery.com/convolutional-layers-for-deep-learning-neural-networks/}{https://machinelearningmastery.com/convolutional-layers-for-deep-learning-neural-networks/}}
\end{frame}

\subsection{Zero Padding}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Convolução}{Zero Padding}
\begin{itemize}
\item A operação de convolução pode impactar na dimensão da imagem resultado, reduzindo-a consideravelmente
\item {\bf Padding} é o processo em que alguns pixels são adicionados ao redor da imagem
\item Assim é possível manter a dimensionalidade na imagem resultante após uma operação de convolução
\end{itemize}

\subsection{Stride}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Convolução}{Stride}
\begin{itemize}

\item {\bf Stride} indica o tamanho do deslocamento ao deslizar o {\it kernel} sobre a imagem

\subsection{Dilatation}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Convolução}{Dilatação}
\begin{itemize}
\item Algumas vezes é interessante ``pular'' alguns valores. Ex. ao se trabalhar com imagens de grande dimensão
\item {\bf Convoluções Dilatadas} permitem que o {\it kernel} aumente o ``alcance de visão'', sem aumentar o número de parâmetros
\item Também há perda de informação, pois nem todos os pixels são processados
\end{itemize}

\subsection{Formula}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Formula}
A fórmula da convolução 
\begin{equation}
Out = \frac{(In - F + 2P)} {S} + 1 \nonumber
\end{equation}
em que
\begin{itemize}
\item $In$ é o tamanho da entrada
\item $F$ é o tamanho do campo receptivo (kernel)
\item $P$ é a quantidade de {\it zero padding} na borda
\item $S$ é o {\it stride}
\end{itemize}

\vspace{1cm}
\tiny{Referência: {\color{blue} \href{https://cs231n.github.io/convolutional-networks/}{Convolutional Neural Networks for Visual Recognition} } }
\end{frame}

\subsection{Pooling}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Pooling}
\begin{itemize}
\item Outra estratégia para reduzir a dimensionalidade de uma imagem de entrada é utilizar camadas de {\bf pooling}
\item São também chamadas de camadas {\it downsampling} or {\it subsampling}
\item Existem três diferentes tipos de {\it pooling} espacial:
\begin{itemize}
\item Max Pooling
\item Average Pooling
\item Sum Pooling
\end{itemize}
\end{itemize}

\vspace{1cm}
\tiny{Referência: {\color{blue} \href{https://towardsdatascience.com/types-of-convolution-kernels-simplified-f040cb307c37}{Types of Convolution Kernels : Simplified} }}
\end{frame}
