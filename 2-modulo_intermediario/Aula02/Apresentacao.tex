\section{Descending Gradient}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Descending Gradient}
\begin{itemize}
\item O {\bf Gradiente Descendente} é um algoritmo de otimização usado para treinar redes neurais.
\item Sua função é encontrar um conjunto de parâmetros que minimizam uma função custo.
\item É usado quando os parâmetros não podem ser calculados analiticamente e devem ser encontrados por um algoritmo de otimização.
\end{itemize}

\vspace{3cm}
\tiny{Referência: \href{https://machinelearningmastery.com/gradient-descent-for-machine-learning/}{https://machinelearningmastery.com/gradient-descent-for-machine-learning/}}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Processo de Treinamento}{Descending Gradient}
\begin{figure}[h]
\includegraphics[width=9cm]{imgs/mlp_gradient_descent3.png}
\end{figure}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Descending Gradient}{Pseudo-Código}
\begin{algorithm}[H]
\caption{Gradiente Descendente Estocástico} 
\label{perceptron}
\Entrada{$\vec{x} = \{ (x_1, ... x_n)\}$ $\vec{y} = \{y_1, ... y_n\}  \}$}
\Saida{$\vec{w}$}
\Inicio{
 %\tcp{Inicializa os pesos}
  Inicializa $\vec{w}$ e $b$ com valores aleatórios; \\
  \Repita {$n$ épocas} {
    \Para {cada amostra ($x_i, y_i$) de ($\vec{x}, \vec{y}$)} {
    	Calcule $\hat{y}_i \leftarrow \vec{w} x_i + b$; \\
        Calcule a Loss: $\mathcal{L}_i \leftarrow L(\hat{y}_i, y_i)$; \\ 
		Calcule o gradiente: $\Delta w \leftarrow - \nabla \mathcal{L}_i (w); \Delta b \leftarrow -  \frac{ \partial \mathcal{L}_i } {\partial b} $ ;\\
 	    \tcp{Atualiza os parâmetros}
		$w \leftarrow w + \alpha \Delta w$ ; $b \leftarrow b + \alpha \Delta b$; \\		
     }
  }
}
\end{algorithm}
\end{frame}

\begin{frame}{Descending Gradient}{Variações}
\begin{itemize}
\item {\bf Gradiente Descendente Estocástico}
\begin{itemize}
\item A atualização dos coeficientes é realizada para cada instância de treinamento.
\item {\it Batch-Size} $= 1$.
\item Na versão verdadeiramente estocástica a amostra extraída é devolvida ao conjunto.
\end{itemize}
\item {\bf Gradiente Descendente em Lote}
\begin{itemize}
\item A atualização dos coeficientes é realizada após o cálculo da {\it Loss} de todas as instâncias de treinamento.
\item {\it Batch-Size} igual ao tamanho do conjunto de treinamento.
\item É chamado {\it Batch Gradient Descent} ou {\it Vanilla Gradient Descent}.
\end{itemize}
\item {\bf Gradiente Descendente em Mini-Lote}
\begin{itemize}
\item A atualização dos coeficientes é realizada após o cálculo da {\it Loss} de um conjunto de instâncias de treinamento.
\item Quanto maior o tamanho do {\it Batch-Size} mais estável é o treinamento.
\end{itemize}
\end{itemize}

\vspace{1cm}
\tiny{Referência: \href{https://sebastianraschka.com/faq/docs/sgd-methods.html}{https://sebastianraschka.com/faq/docs/sgd-methods.html}}
\end{frame}

\section{Multi-Class Problem}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Multi-Class Problem}
\begin{itemize}
\item Em problemas de classificação binária utiliza-se a função {\it Sigmoid} como função de ativação 
\item A função Sigmoide produz como saída um valor entre zero e um.
\item Como função Loss, utiliza-se a {\it Binary Cross Entropy}.
\end{itemize}
\end{frame}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Multi-Class Problem}
\begin{itemize}
\item Para problemas {\bf multiclasse}, utiliza-se a função {\it Softmax} como função de ativação.
\item A função Softmax estende o conceito da função Sigmoid para problemas multiclasse.
\item Softmax atribui uma probabilidade (entre 0 e 1) para cada classe sendo que o somatório é igual a 1.
\end{itemize}
\vspace{1cm}
\tiny{Referência: \href{https://developers.google.com/machine-learning/crash-course/multi-class-neural-networks/softmax}{Multi-Class Neural Networks: Softmax}}

\end{frame}

\begin{frame}{Multi-Class Problem}{Categorical Cross Entropy}
\begin{itemize}
\item A função Loss {\it Categorical Cross Entropy} é adequada para problemas de classificação multi-classe.
\item Seu cálculo é dado pela seguinte fórmula:
\end{itemize}

\begin{equation}
J(z) = \frac{-1}{N} \sum_j^{M} \sum_i^{N} y_{ij} \log (\hat{y_{ij}}) + (1 - y_{ij}) \log (1 - \hat{y_{ij}}) \nonumber	
\end{equation}
\end{frame}

\begin{frame}{Learning Rate}{Exemplo Keras}
\begin{itemize}
\item A {\bf taxa de aprendizagem} é um hiperparâmetro configurável usado no treinamento de redes neurais que possui um pequeno valor (entre 0,0 e 1,0).
\item Valores menores requerem mais {\it épocas} de treinamento devido às mudanças menores feitas nos pesos a cada atualização.
\item Valores maiores resultam em mudanças rápidas e requerem menos períodos de treinamento.
\end{itemize}
\end{frame}

\begin{itemize}
\item O desafio de treinar redes neurais de aprendizado profundo envolve a seleção cuidadosa da taxa de aprendizado.	
\item Uma taxa de aprendizado muito grande pode fazer com que o modelo convirja muito rapidamente para uma solução abaixo do ideal,
\item Um valor muito pequeno pode fazer com que o processo de aprendizado pare.
\end{itemize}
