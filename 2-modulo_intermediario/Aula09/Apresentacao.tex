\section{Introdução}
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\begin{frame}{Introdução}{Style Transfer}
\begin{itemize}
\item A transferência de estilo neural é uma técnica de otimização usada para pegar duas imagens.
\item Uma imagem de conteúdo e uma imagem de referência de estilo.
\item As imagens são misturadas para que a imagem de saída se pareça com a imagem de conteúdo.
\end{itemize}
\end{frame}

\begin{frame}{Introdução}{Style Transfer}
\begin{itemize}
\item Isso é implementado otimizando a imagem de saída para corresponder:
\begin{itemize}
\item às estatísticas de conteúdo da imagem de conteúdo, e
\item às estatísticas de estilo da imagem de referência de estilo.
\end{itemize}  
\item  Essas estatísticas são extraídas das imagens usando uma rede convolucional.
\end{itemize}
\end{frame}

\begin{frame}{Introdução}{Style Transfer}
\begin{itemize}
\item Conceito de ``reconstrução de estilo'': ao invés de combinar as ativações, combina a correlação das ativações.
\item O recurso de correlação é chamado de matriz de Gram: 
\begin{itemize}
\item produto de ponto entre a matriz de ativação de característica vetorial e sua transposição.
\end{itemize} 
\end{itemize}
\end{frame}
